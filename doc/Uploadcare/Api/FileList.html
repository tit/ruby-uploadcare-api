def initialize api, results
  @api = api
  @files = results['results'].map do |obj|
    Api::File.new(api, obj)
  end
  @page = results['page'].to_i
  @per_page = results['per_page'].to_i
  @total = results['total'].to_i
  @pages = results['pages'].to_i
end
